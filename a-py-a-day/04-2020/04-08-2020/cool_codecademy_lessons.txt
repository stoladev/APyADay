# HOW TO KEEP NESTING, AND HOW THIS MAKES SUCH LOGICAL SENSE
def over_nine_thousand(lst):
    total = 0
    while total <= 9000:
        for num in lst:
            if total <= 9000:
                total += num
            elif total > 9000:
                return total
            elif not total:
                return 0
        break
    return total


print(over_nine_thousand([8000, 900]))
print(over_nine_thousand([8000, 900, 120, 5000]))

# THE IMPORTANCE OF SETTING A VARIABLE (=) VS A MEANINGLESS BOOLEAN (==)
def max_num(nums):
    largest_number = nums[0]
    for num in nums:
        if num > largest_number:
            largest_number = num # I SET THIS TO == num AND IT TOOK ME 30 MINUTES TO FIGURE OUT THE PROBLEM
            continue
        else:
            continue
    return largest_number


print(max_num([50, -10, 0, 75, 20]))
